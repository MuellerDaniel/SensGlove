FUNCTION cel(kc, p, c, s) \\
\hspace*{1cm}\=\hspace*{1cm}\= \kill
\> IF (kc = 0) THEN \\
\> \> cel = NAN \\
\> \> EXIT FUNCTION\\
\> ENDIF\\
\> errtol = .000001\\
\> k = ABS(kc)\\
\> pp = p\\
\> cc = c\\
\> ss = s\\
\> em = 1.\\
\> IF (p $ \geq $ 0) THEN\\
\> \> pp = SQR(p)\\
\> \> ss = s/pp \\
\> ELSE\\
\> \> f = kc*kc\\
